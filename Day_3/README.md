# Types of Function Arguments in Python : 

# 1. Positional Arguments : 

      Syntax : 
          def FunctionName(arg1 , arg2):
                statements

# 2. Keyword Arguments : 

      Syntax : 
          def FunctionName(parameter = value):
                  statements

# 3. Default Arguments : 

      Syntax : 
          def FunctionName(arg1 , arg2 = default):
                  statements

# 4. Variable Number of Arguments(*args):

      Syntax : 
          def FunctionName(*args):
                  statements
